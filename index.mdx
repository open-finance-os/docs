<Tabs>
  <Tab title="Get Account Balance">
    ```javascript
    // Copy this working example from the hackathon starter
    const { getAccessToken } = require('./examples/auth');
    const axios = require('axios');

    async function getAccountBalance() {
      const { token } = await getAccessToken();

      const response = await axios.get(
        `${process.env.OPENFINANCE_BASE_URL}/accounts`,
        {
          headers: { 'Authorization': `Bearer ${token}` }
        }
      );

      console.log('Account Balance:', response.data.data[0].balance);
    }
    ```
