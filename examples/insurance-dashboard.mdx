### **Step 1: Basic API Setup**
```javascript
// Simple insurance API client
const INSURANCE_API = 'https://sandbox.openfinance.ae/api/v1/insurance';

const insuranceClient = {
  get: async (endpoint, token) => {
    const response = await fetch(`${INSURANCE_API}${endpoint}`, {
      headers: {
        'Authorization': `Bearer ${token}`,
        'Accept': 'application/json'
      }
    });

    if (!response.ok) {
      throw new Error(`Insurance API Error: ${response.status}`);
    }

    return response.json();
  },

  post: async (endpoint, data, token) => {
    const response = await fetch(`${INSURANCE_API}${endpoint}`, {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${token}`,
        'Content-Type': 'application/json'
      },
      body: JSON.stringify(data)
    });

    return response.json();
  }
};
```

### **Step 2: Policy Dashboard**
```javascript
// Display user's insurance policies
const PolicyDashboard = ({ token }) => {
  const [policies, setPolicies] = useState([]);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    const fetchPolicies = async () => {
      try {
        const response = await insuranceClient.get('/policies', token);
        setPolicies(response.policies);
      } catch (error) {
        console.error('Error fetching policies:', error);
      } finally {
        setLoading(false);
      }
    };

    fetchPolicies();
  }, [token]);

  const activePolicies = policies.filter(p => p.status === 'active');
  const totalCoverage = activePolicies.reduce((sum, p) => sum + p.coverageAmount, 0);

  if (loading) return <div>Loading policies...</div>;

  return (
    <div className="policy-dashboard">
      <div className="dashboard-stats">
        <div className="stat-card">
          <h3>Active Policies</h3>
          <p className="stat-number">{activePolicies.length}</p>
        </div>
        <div className="stat-card">
          <h3>Total Coverage</h3>
          <p className="stat-number">AED {totalCoverage.toLocaleString()}</p>
        </div>
      </div>

      <div className="policies-grid">
        {activePolicies.map(policy => (
          <PolicyCard key={policy.id} policy={policy} />
        ))}
      </div>
    </div>
  );
};
```

### **Step 3: Claims Submission Form**
```javascript
// Simple but effective claims submission
const ClaimsForm = ({ token }) => {
  const [claim, setClaim] = useState({
    policyId: '',
    incidentDate: '',
    description: '',
    estimatedAmount: ''
  });
  const [submitting, setSubmitting] = useState(false);

  const handleSubmit = async (e) => {
    e.preventDefault();
    setSubmitting(true);

    try {
      const response = await insuranceClient.post('/claims', {
        policyId: claim.policyId,
        incidentDate: claim.incidentDate,
        description: claim.description,
        claimAmount: parseFloat(claim.estimatedAmount),
        status: 'submitted',
        submissionDate: new Date().toISOString()
      }, token);

      if (response.success) {
        alert('Claim submitted successfully!');
        // Reset form
        setClaim({ policyId: '', incidentDate: '', description: '', estimatedAmount: '' });
      }
    } catch (error) {
      console.error('Claim submission failed:', error);
      alert('Failed to submit claim. Please try again.');
    } finally {
      setSubmitting(false);
    }
  };

  return (
    <form onSubmit={handleSubmit} className="claims-form">
      <select
        value={claim.policyId}
        onChange={(e) => setClaim({...claim, policyId: e.target.value})}
        required
      >
        <option value="">Select Policy</option>
        <option value="health-001">Health Insurance - Premium</option>
        <option value="auto-002">Auto Insurance - Comprehensive</option>
        <option value="home-003">Home Insurance - Standard</option>
      </select>

      <input
        type="date"
        placeholder="Incident Date"
        value={claim.incidentDate}
        onChange={(e) => setClaim({...claim, incidentDate: e.target.value})}
        required
      />

      <textarea
        placeholder="Describe the incident..."
        value={claim.description}
        onChange={(e) => setClaim({...claim, description: e.target.value})}
        required
      />

      <input
        type="number"
        placeholder="Estimated Amount (AED)"
        value={claim.estimatedAmount}
        onChange={(e) => setClaim({...claim, estimatedAmount: e.target.value})}
        required
      />

      <button type="submit" disabled={submitting}>
        {submitting ? 'Submitting...' : 'Submit Claim'}
      </button>
    </form>
  );
};
```

### **Step 4: Simple Quote Calculator**
```javascript
// Basic insurance quote calculator
const QuoteCalculator = () => {
  const [quoteData, setQuoteData] = useState({
    insuranceType: '',
    coverageAmount: '',
    age: '',
    location: ''
  });
  const [quote, setQuote] = useState(null);
  const [calculating, setCalculating] = useState(false);

  const calculateQuote = async () => {
    setCalculating(true);

    try {
      // Simple quote calculation simulation
      const baseRates = {
        health: 1200,
        auto: 2500,
        home: 800
      };

      const baseRate = baseRates[quoteData.insuranceType] || 1000;
      const ageMultiplier = quoteData.age > 50 ? 1.2 : 1.0;
      const coverageMultiplier = parseFloat(quoteData.coverageAmount) / 100000;

      const annualPremium = baseRate * ageMultiplier * coverageMultiplier;

      setQuote({
        annualPremium: annualPremium.toFixed(2),
        monthlyPremium: (annualPremium / 12).toFixed(2),
        coverageAmount: quoteData.coverageAmount,
        policyType: quoteData.insuranceType
      });
    } catch (error) {
      console.error('Quote calculation failed:', error);
    } finally {
      setCalculating(false);
    }
  };

  return (
    <div className="quote-calculator">
      <h3>Get Instant Quote</h3>

      <select
        value={quoteData.insuranceType}
        onChange={(e) => setQuoteData({...quoteData, insuranceType: e.target.value})}
      >
        <option value="">Select Insurance Type</option>
        <option value="health">Health Insurance</option>
        <option value="auto">Auto Insurance</option>
        <option value="home">Home Insurance</option>
      </select>

      <input
        type="number"
        placeholder="Coverage Amount (AED)"
        value={quoteData.coverageAmount}
        onChange={(e) => setQuoteData({...quoteData, coverageAmount: e.target.value})}
      />

      <input
        type="number"
        placeholder="Your Age"
        value={quoteData.age}
        onChange={(e) => setQuoteData({...quoteData, age: e.target.value})}
      />

      <button onClick={calculateQuote} disabled={calculating}>
        {calculating ? 'Calculating...' : 'Calculate Quote'}
      </button>

      {quote && (
        <div className="quote-result">
          <h4>Your Quote</h4>
          <p>Coverage: AED {parseInt(quote.coverageAmount).toLocaleString()}</p>
          <p>Annual Premium: AED {quote.annualPremium}</p>
          <p>Monthly Premium: AED {quote.monthlyPremium}</p>
        </div>
      )}
    </div>
  );
};
```
