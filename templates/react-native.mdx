    expo start

## ğŸ” Security Implementation

### **Biometric Authentication**
```typescript
// services/auth/BiometricAuth.ts
import TouchID from 'react-native-touch-id';
import { Alert } from 'react-native';

export class BiometricAuth {
  static async isSupported(): Promise<boolean> {
    try {
      const biometryType = await TouchID.isSupported();
      return !!biometryType;
    } catch {
      return false;
    }
  }

  static async authenticate(): Promise<boolean> {
    try {
      await TouchID.authenticate('Access your banking app', {
        title: 'Biometric Authentication',
        subtitle: 'Scan your biometric to continue',
        description: 'This app uses biometric authentication for security',
        fallbackLabel: 'Use Passcode',
        cancelLabel: 'Cancel',
        passcodeFallback: true,
      });
      return true;
    } catch (error) {
      if (error.name === 'LAErrorUserFallback') {
        // User chose to use passcode
        return this.authenticateWithPasscode();
      }
      Alert.alert('Authentication Failed', 'Please try again');
      return false;
    }
  }

  private static async authenticateWithPasscode(): Promise<boolean> {
    // Implement passcode authentication
    return new Promise((resolve) => {
      // Show passcode modal
      resolve(true);
    });
  }
}
```

### **Secure Storage**
```typescript
// services/storage/SecureStorage.ts
import Keychain from 'react-native-keychain';

export class SecureStorage {
  static async setItem(key: string, value: string): Promise<void> {
    await Keychain.setInternetCredentials(
      key,
      'user',
      value,
      {
        accessControl: Keychain.ACCESS_CONTROL.BIOMETRY_CURRENT_SET,
        accessGroup: 'group.com.yourapp.shared',
        authenticatePrompt: 'Authenticate to access secure data',
      }
    );
  }

  static async getItem(key: string): Promise<string | null> {
    try {
      const credentials = await Keychain.getInternetCredentials(key);
      return credentials ? credentials.password : null;
    } catch {
      return null;
    }
  }

  static async removeItem(key: string): Promise<void> {
    await Keychain.resetInternetCredentials(key);
  }
}
```

### **Account Overview Screen**
```typescript
// screens/banking/AccountOverviewScreen.tsx
import React, { useEffect, useState } from 'react';
import {
  View,
  FlatList,
  RefreshControl,
  StyleSheet,
} from 'react-native';
import { useAppDispatch, useAppSelector } from '../../hooks/redux';
import { fetchAccounts } from '../../store/slices/accountsSlice';
import { AccountCard } from '../../components/banking/AccountCard';
import { LoadingSpinner } from '../../components/common/LoadingSpinner';

export const AccountOverviewScreen: React.FC = () => {
  const dispatch = useAppDispatch();
  const { accounts, loading, error } = useAppSelector(state => state.accounts);
  const [refreshing, setRefreshing] = useState(false);

  useEffect(() => {
    dispatch(fetchAccounts());
  }, [dispatch]);

  const onRefresh = async () => {
    setRefreshing(true);
    await dispatch(fetchAccounts());
    setRefreshing(false);
  };

  const renderAccount = ({ item }: { item: Account }) => (
    <AccountCard
      account={item}
      onPress={() => navigateToAccountDetails(item.id)}
    />
  );

  if (loading && !refreshing) {
    return <LoadingSpinner />;
  }

  return (
    <View style={styles.container}>
      <FlatList
        data={accounts}
        renderItem={renderAccount}
        keyExtractor={item => item.id}
        refreshControl={
          <RefreshControl
            refreshing={refreshing}
            onRefresh={onRefresh}
            tintColor="#007AFF"
          />
        }
        contentContainerStyle={styles.listContainer}
        showsVerticalScrollIndicator={false}
      />
    </View>
  );
};

const styles = StyleSheet.create({
  container: {
    flex: 1,
    backgroundColor: '#F5F5F5',
  },
  listContainer: {
    padding: 16,
  },
});
```

### **Transaction History with Infinite Scroll**
```typescript
// components/banking/TransactionList.tsx
import React, { useCallback, useState } from 'react';
import {
  FlatList,
  ActivityIndicator,
  View,
  Text,
} from 'react-native';
import { Transaction } from '../../types/banking';
import { TransactionItem } from './TransactionItem';

interface Props {
  accountId: string;
  transactions: Transaction[];
  onLoadMore: () => void;
  hasMore: boolean;
  loading: boolean;
}

export const TransactionList: React.FC<Props> = ({
  accountId,
  transactions,
  onLoadMore,
  hasMore,
  loading,
}) => {
  const renderTransaction = useCallback(
    ({ item }: { item: Transaction }) => (
      <TransactionItem transaction={item} />
    ),
    []
  );

  const renderFooter = () => {
    if (!loading) return null;
    return (
      <View style={styles.footer}>
        <ActivityIndicator size="small" color="#007AFF" />
        <Text style={styles.loadingText}>Loading more transactions...</Text>
      </View>
    );
  };

  const handleEndReached = () => {
    if (hasMore && !loading) {
      onLoadMore();
    }
  };

  return (
    <FlatList
      data={transactions}
      renderItem={renderTransaction}
      keyExtractor={item => item.id}
      onEndReached={handleEndReached}
      onEndReachedThreshold={0.1}
      ListFooterComponent={renderFooter}
      showsVerticalScrollIndicator={false}
    />
  );
};
```

## ğŸ“Š State Management

### **Redux Store Configuration**
```typescript
// store/index.ts
import { configureStore } from '@reduxjs/toolkit';
import { persistStore, persistReducer } from 'redux-persist';
import AsyncStorage from '@react-native-async-storage/async-storage';
import { combineReducers } from 'redux';

import authSlice from './slices/authSlice';
import accountsSlice from './slices/accountsSlice';
import transactionsSlice from './slices/transactionsSlice';
import settingsSlice from './slices/settingsSlice';

const persistConfig = {
  key: 'root',
  storage: AsyncStorage,
  whitelist: ['auth', 'settings'], // Only persist these slices
  blacklist: ['accounts', 'transactions'], // Don't persist sensitive data
};

const rootReducer = combineReducers({
  auth: authSlice,
  accounts: accountsSlice,
  transactions: transactionsSlice,
  settings: settingsSlice,
});

const persistedReducer = persistReducer(persistConfig, rootReducer);

export const store = configureStore({
  reducer: persistedReducer,
  middleware: (getDefaultMiddleware) =>
    getDefaultMiddleware({
      serializableCheck: {
        ignoredActions: ['persist/PERSIST', 'persist/REHYDRATE'],
      },
    }),
});

export const persistor = persistStore(store);
export type RootState = ReturnType<typeof store.getState>;
export type AppDispatch = typeof store.dispatch;
```
